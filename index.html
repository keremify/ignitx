01 NON-FOLLOWERS DETECTOR
FOLLOW
Scans your entire following list and compares it against your followers. Outputs a clean, sorted list of every account that doesn't follow you back — with their usernames and profile links.
01 › Go to instagram.com and make sure you're logged in
02 › Open the console and paste the script below
03 › Wait for the scan to finish — a list will appear in the console
javascript 
// ── INSTATOOL: Non-Followers Detector ──────────────────────
// Finds everyone you follow who does NOT follow you back.

(async () => {
  const fetchList = async (type) => {
    let list = [], after = "", hasNext = true;
    const userId = window._sharedData?.config?.viewerId ||
      document.cookie.match(/ds_user_id=(\d+)/)?.[1];
    if (!userId) { console.error("❌ Could not get user ID. Are you logged in?"); return []; }
    while (hasNext) {
      const url = `https://www.instagram.com/api/v1/friendships/${userId}/${type}/?count=200${after ? "&max_id=" + after : ""}`;
      const res = await fetch(url).then(r => r.json());
      list.push(...(res.users || []));
      hasNext = !!res.next_max_id;
      after = res.next_max_id || "";
      await new Promise(r => setTimeout(r, 500));
    }
    return list;
  };

  console.log("%c⏳ Fetching your following list...", "color:#00ff88");
  const following = await fetchList("following");
  console.log("%c⏳ Fetching your followers list...", "color:#00ff88");
  const followers = await fetchList("followers");

  const followerSet = new Set(followers.map(u => u.pk));
  const notFollowingBack = following.filter(u => !followerSet.has(u.pk));

  console.log(`%c\n📊 RESULTS: ${notFollowingBack.length} accounts not following you back:\n`, "color:#ffcc00;font-weight:bold");
  notFollowingBack.forEach((u, i) => {
    console.log(`%c${i+1}. @${u.username} — https://instagram.com/${u.username}`, "color:#c8f5d8");
  });
  console.log(`\n%c✅ Done. Total following: ${following.length} | Followers: ${followers.length} | Not following back: ${notFollowingBack.length}`, "color:#00ff88;font-weight:bold");
})();
02 FOLLOW A USER'S FOLLOWINGS
FOLLOW
Enter a target username and this script will bulk follow everyone that user follows. Built-in delays keep the action safe and human-like to avoid Instagram rate limits.
01 › Replace TARGET_USERNAME in the script with the account you want to clone follows from
02 › Paste into the console while logged in on Instagram
03 › The script follows ~200 users with safe delays — run again for more
javascript 
// ── INSTATOOL: Follow a User's Followings ──────────────────
// ⚠ Replace with your target username below:
const TARGET_USERNAME = "TARGET_USERNAME";
const DELAY_MS = 3000; // 3 seconds between each follow (safe)

(async () => {
  const sleep = ms => new Promise(r => setTimeout(r, ms));

  // Step 1: Get target user ID
  console.log("%c⏳ Looking up user...", "color:#00ff88");
  const userRes = await fetch(`https://www.instagram.com/api/v1/users/web_profile_info/?username=${TARGET_USERNAME}`,
    { headers: { "x-ig-app-id": "936619743392459" } }).then(r => r.json());
  const targetId = userRes?.data?.user?.id;
  if (!targetId) { console.error("❌ User not found. Check the username."); return; }
  console.log(`%c✅ Found @${TARGET_USERNAME} (ID: ${targetId})`, "color:#00ff88");

  // Step 2: Fetch their following list
  let toFollow = [], after = "", hasNext = true;
  while (hasNext) {
    const res = await fetch(`https://www.instagram.com/api/v1/friendships/${targetId}/following/?count=200${after ? "&max_id=" + after : ""}`).then(r => r.json());
    toFollow.push(...(res.users || []));
    hasNext = !!res.next_max_id;
    after = res.next_max_id || "";
    await sleep(800);
  }
  console.log(`%c📋 Found ${toFollow.length} users to follow`, "color:#ffcc00");

  // Step 3: Follow each user with a delay
  const csrfToken = document.cookie.match(/csrftoken=([^;]+)/)?.[1];
  let followed = 0, skipped = 0;
  for (const user of toFollow) {
    const res = await fetch(`https://www.instagram.com/api/v1/friendships/create/${user.pk}/`, {
      method: "POST",
      headers: { "x-csrftoken": csrfToken, "content-type": "application/x-www-form-urlencoded" }
    }).then(r => r.json());
    if (res.friendship_status) {
      followed++;
      console.log(`%c✓ Followed @${user.username} [${followed}/${toFollow.length}]`, "color:#00ff88");
    } else { skipped++; }
    await sleep(DELAY_MS);
  }
  console.log(`%c\n✅ Done! Followed: ${followed} | Skipped: ${skipped}`, "color:#ffcc00;font-weight:bold");
})();
Following too many accounts too quickly may trigger Instagram's rate limiter. The default 3-second delay is recommended. Do not run this more than once per hour.
03 MASS UNFOLLOW NON-FOLLOWERS
FOLLOW
Automatically unfollows everyone who doesn't follow you back. Supports a whitelist — add usernames you never want to unfollow (friends, brands) to keep them safe.
01 › Add any usernames you want to keep to the WHITELIST array
02 › Set MAX_UNFOLLOW to cap how many to unfollow in one session
03 › Paste and run — sit back and watch
javascript 
// ── INSTATOOL: Mass Unfollow Non-Followers ─────────────────
const WHITELIST = ["friend1", "brand_i_like"]; // never unfollow these
const MAX_UNFOLLOW = 50;  // max per session — stay under 100/hour total
const DELAY_MS = 3500;

(async () => {
  const sleep = ms => new Promise(r => setTimeout(r, ms));
  const userId = document.cookie.match(/ds_user_id=(\d+)/)?.[1];
  const csrfToken = document.cookie.match(/csrftoken=([^;]+)/)?.[1];
  if (!userId) { console.error("❌ Not logged in."); return; }

  const getList = async (type) => {
    let all = [], cursor = "", more = true;
    while (more) {
      const r = await fetch(`https://www.instagram.com/api/v1/friendships/${userId}/${type}/?count=200${cursor ? "&max_id=" + cursor : ""}`).then(x => x.json());
      all.push(...(r.users || []));
      more = !!r.next_max_id; cursor = r.next_max_id || "";
      await sleep(600);
    }
    return all;
  };

  console.log("%c⏳ Loading lists...", "color:#00ff88");
  const [following, followers] = await Promise.all([getList("following"), getList("followers")]);
  const followerIds = new Set(followers.map(u => u.pk));
  const targets = following
    .filter(u => !followerIds.has(u.pk) && !WHITELIST.includes(u.username))
    .slice(0, MAX_UNFOLLOW);

  console.log(`%c🗑 Unfollowing ${targets.length} accounts...`, "color:#ff4466");
  let count = 0;
  for (const u of targets) {
    await fetch(`https://www.instagram.com/api/v1/friendships/destroy/${u.pk}/`, {
      method: "POST",
      headers: { "x-csrftoken": csrfToken, "content-type": "application/x-www-form-urlencoded" }
    });
    count++;
    console.log(`%c✗ Unfollowed @${u.username} [${count}/${targets.length}]`, "color:#ff4466");
    await sleep(DELAY_MS);
  }
  console.log(`%c✅ Session complete. Unfollowed ${count} accounts.`, "color:#ffcc00;font-weight:bold");
})();
04 EXPORT FOLLOWERS / FOLLOWING TO CSV
FOLLOW
Downloads your complete followers and following lists as a CSV file to your computer. Includes username, full name, and profile URL for every account.
01 › Choose to export "followers" or "following" by editing the EXPORT_TYPE variable
02 › Paste the script — a .csv file will automatically download when done
javascript 
// ── INSTATOOL: Export to CSV ────────────────────────────────
const EXPORT_TYPE = "followers"; // "followers" or "following"

(async () => {
  const userId = document.cookie.match(/ds_user_id=(\d+)/)?.[1];
  let list = [], cursor = "", more = true;
  console.log(`%c⏳ Fetching ${EXPORT_TYPE}...`, "color:#00ff88");
  while (more) {
    const r = await fetch(`https://www.instagram.com/api/v1/friendships/${userId}/${EXPORT_TYPE}/?count=200${cursor ? "&max_id=" + cursor : ""}`).then(x => x.json());
    list.push(...(r.users || []));
    more = !!r.next_max_id; cursor = r.next_max_id || "";
    await new Promise(r => setTimeout(r, 500));
  }

  const csv = [
    "username,full_name,profile_url,is_private,followers",
    ...list.map(u => `@${u.username},"${u.full_name || ""}",https://instagram.com/${u.username},${u.is_private},${u.follower_count || ""}`)
  ].join("\n");

  const blob = new Blob([csv], { type: "text/csv" });
  const a = Object.assign(document.createElement("a"), {
    href: URL.createObjectURL(blob),
    download: `instagram_${EXPORT_TYPE}_${new Date().toISOString().slice(0,10)}.csv`
  });
  document.body.appendChild(a); a.click(); a.remove();
  console.log(`%c✅ Exported ${list.length} ${EXPORT_TYPE} to CSV!`, "color:#00ff88;font-weight:bold");
})();
05 LIKE TAGGED POSTS OF A USER
ENGAGE
Automatically likes all posts that a specific user is tagged in. Great for boosting engagement with collaborators or showing support. Set a limit to control how many posts to like.
01 › Set TARGET_USERNAME to the account whose tagged posts you want to like
02 › Set MAX_LIKES to limit how many posts to like in one session
javascript 
// ── INSTATOOL: Like Tagged Posts of a User ─────────────────
const TARGET_USERNAME = "TARGET_USERNAME";
const MAX_LIKES = 30;
const DELAY_MS = 4000;

(async () => {
  const sleep = ms => new Promise(r => setTimeout(r, ms));
  const csrfToken = document.cookie.match(/csrftoken=([^;]+)/)?.[1];

  // Get target user ID
  const uRes = await fetch(`https://www.instagram.com/api/v1/users/web_profile_info/?username=${TARGET_USERNAME}`,
    { headers: { "x-ig-app-id": "936619743392459" } }).then(r => r.json());
  const targetId = uRes?.data?.user?.id;
  if (!targetId) { console.error("❌ User not found."); return; }
  console.log(`%c✅ Found @${TARGET_USERNAME}`, "color:#00ff88");

  // Fetch tagged posts
  let posts = [], cursor = "", more = true;
  while (more && posts.length < MAX_LIKES) {
    const r = await fetch(`https://www.instagram.com/api/v1/usertags/${targetId}/feed/?count=20${cursor ? "&max_id=" + cursor : ""}`).then(x => x.json());
    posts.push(...(r.items || []));
    more = !!r.next_max_id; cursor = r.next_max_id || "";
    await sleep(600);
  }
  posts = posts.slice(0, MAX_LIKES);
  console.log(`%c❤ Liking ${posts.length} tagged posts...`, "color:#ff4466");

  let liked = 0;
  for (const post of posts) {
    if (post.has_liked) { console.log(`%c⟳ Already liked post ${post.pk}`, "color:#5a8a68"); continue; }
    await fetch(`https://www.instagram.com/api/v1/media/${post.pk}/like/`, {
      method: "POST",
      headers: { "x-csrftoken": csrfToken, "content-type": "application/x-www-form-urlencoded" }
    });
    liked++;
    console.log(`%c❤ Liked post ${liked}/${posts.length}`, "color:#ff88cc");
    await sleep(DELAY_MS);
  }
  console.log(`%c✅ Done! Liked ${liked} posts.`, "color:#ffcc00;font-weight:bold");
})();
06 LIKE POSTS BY HASHTAG
ENGAGE
Enter a hashtag and like the most recent posts under it. Perfect for niche engagement — liking posts in your space attracts targeted followers back to your profile.
01 › Set HASHTAG (without the # symbol)
02 › Set MAX_LIKES and run — it targets recent posts first
javascript 
// ── INSTATOOL: Like Posts by Hashtag ───────────────────────
const HASHTAG = "streetphotography"; // no # symbol
const MAX_LIKES = 20;
const DELAY_MS = 4500;

(async () => {
  const sleep = ms => new Promise(r => setTimeout(r, ms));
  const csrfToken = document.cookie.match(/csrftoken=([^;]+)/)?.[1];

  console.log(`%c⏳ Fetching posts for #${HASHTAG}...`, "color:#00ff88");
  const feedRes = await fetch(`https://www.instagram.com/api/v1/tags/${HASHTAG}/sections/?tab=recent&count=50`, {
    method: "POST",
    headers: { "x-csrftoken": csrfToken, "content-type": "application/x-www-form-urlencoded" }
  }).then(r => r.json());

  const posts = [];
  (feedRes.sections || []).forEach(section => {
    (section.layout_content?.medias || []).forEach(m => posts.push(m.media));
  });

  const targets = posts.filter(p => !p.has_liked).slice(0, MAX_LIKES);
  console.log(`%c❤ Liking ${targets.length} posts under #${HASHTAG}...`, "color:#ff88cc");

  let count = 0;
  for (const post of targets) {
    await fetch(`https://www.instagram.com/api/v1/media/${post.pk}/like/`, {
      method: "POST",
      headers: { "x-csrftoken": csrfToken, "content-type": "application/x-www-form-urlencoded" }
    });
    count++;
    console.log(`%c❤ [${count}/${targets.length}] Liked by @${post.user?.username}`, "color:#ff88cc");
    await sleep(DELAY_MS);
  }
  console.log(`%c✅ Done! ${count} posts liked under #${HASHTAG}`, "color:#ffcc00;font-weight:bold");
})();
07 ENGAGEMENT RATE CALCULATOR
ANALYTICS
Calculates the true engagement rate of any public account based on their last 12 posts. Outputs avg likes, avg comments, follower count, and final ER% — useful for vetting collaboration partners.
01 › Set TARGET_USERNAME to any public account
02 › Results show avg likes, comments, and engagement rate %
javascript 
// ── INSTATOOL: Engagement Rate Calculator ──────────────────
const TARGET_USERNAME = "TARGET_USERNAME";

(async () => {
  console.log(`%c📊 Analyzing @${TARGET_USERNAME}...`, "color:#00ff88");

  // Get user data
  const uRes = await fetch(`https://www.instagram.com/api/v1/users/web_profile_info/?username=${TARGET_USERNAME}`,
    { headers: { "x-ig-app-id": "936619743392459" } }).then(r => r.json());
  const user = uRes?.data?.user;
  if (!user) { console.error("❌ User not found or account is private."); return; }

  const followerCount = user.edge_followed_by.count;
  const posts = user.edge_owner_to_timeline_media.edges.slice(0, 12).map(e => e.node);

  if (posts.length === 0) { console.warn("⚠ No posts found."); return; }

  const totalLikes = posts.reduce((s, p) => s + (p.edge_media_preview_like?.count || 0), 0);
  const totalComments = posts.reduce((s, p) => s + (p.edge_media_to_comment?.count || 0), 0);
  const avgLikes = Math.round(totalLikes / posts.length);
  const avgComments = Math.round(totalComments / posts.length);
  const engRate = ((totalLikes + totalComments) / (posts.length * followerCount) * 100).toFixed(2);

  const rating = engRate > 6 ? "🔥 Excellent" : engRate > 3 ? "✅ Good" : engRate > 1 ? "⚠ Average" : "❌ Low";
  console.log(`%c
┌─────────────────────────────────────┐
│  📊 ENGAGEMENT REPORT: @${TARGET_USERNAME}
├─────────────────────────────────────┤
│  Followers:      ${followerCount.toLocaleString()}
│  Posts analyzed: ${posts.length}
│  Avg Likes:      ${avgLikes.toLocaleString()}
│  Avg Comments:   ${avgComments.toLocaleString()}
│  Engagement Rate: ${engRate}%  ${rating}
└─────────────────────────────────────┘`, "color:#00ff88;font-family:monospace");
})();
08 GHOST FOLLOWER FINDER
ANALYTICS
Identifies followers who have never engaged with any of your last 12 posts. Returns a list sorted by most likely to be ghost/bot accounts — useful for cleaning your audience.
01 › Paste the script while logged in — no setup needed
02 › It will cross-reference your follower list against your post likers
javascript 
// ── INSTATOOL: Ghost Follower Finder ───────────────────────
// Checks which followers have NEVER liked any of your last 12 posts

(async () => {
  const sleep = ms => new Promise(r => setTimeout(r, ms));
  const userId = document.cookie.match(/ds_user_id=(\d+)/)?.[1];
  if (!userId) { console.error("❌ Not logged in."); return; }

  // Get own username
  const meRes = await fetch(`https://www.instagram.com/api/v1/users/${userId}/info/`).then(r => r.json());
  const myUsername = meRes?.user?.username;

  // Get my posts
  console.log("%c⏳ Fetching your recent posts...", "color:#00ff88");
  const profileRes = await fetch(`https://www.instagram.com/api/v1/users/web_profile_info/?username=${myUsername}`,
    { headers: { "x-ig-app-id": "936619743392459" } }).then(r => r.json());
  const posts = profileRes?.data?.user?.edge_owner_to_timeline_media?.edges?.slice(0, 12) || [];

  // Collect all engagers (likers) across posts
  const engagers = new Set();
  for (const { node: post } of posts) {
    const likers = await fetch(`https://www.instagram.com/api/v1/media/${post.id}/likers/`).then(r => r.json());
    (likers.users || []).forEach(u => engagers.add(u.pk));
    await sleep(400);
  }
  console.log(`%c📬 Unique engagers across last ${posts.length} posts: ${engagers.size}`, "color:#00ff88");

  // Get followers and find ghosts
  console.log("%c⏳ Loading followers...", "color:#00ff88");
  let followers = [], cursor = "", more = true;
  while (more) {
    const r = await fetch(`https://www.instagram.com/api/v1/friendships/${userId}/followers/?count=200${cursor ? "&max_id=" + cursor : ""}`).then(x => x.json());
    followers.push(...(r.users || []));
    more = !!r.next_max_id; cursor = r.next_max_id || "";
    await sleep(500);
  }

  const ghosts = followers.filter(u => !engagers.has(u.pk));
  console.log(`%c\n👻 GHOST FOLLOWERS (${ghosts.length} of ${followers.length}):\n`, "color:#ff88cc;font-weight:bold");
  ghosts.forEach((u, i) => console.log(`%c${i+1}. @${u.username}`, "color:#c8f5d8"));
  console.log(`%c\nGhost rate: ${((ghosts.length / followers.length) * 100).toFixed(1)}%`, "color:#ffcc00");
})();
This tool defines "ghost" as anyone who didn't like your last 12 posts. Some real followers simply browse without liking — take this as a signal, not a verdict.
09 BANNED HASHTAG CHECKER
ANALYTICS
Paste a list of hashtags and this script checks each one against Instagram's feed to detect shadow-banned or blocked tags. Using banned hashtags kills your post reach silently.
01 › Replace the HASHTAGS array with your own list
02 › The script will mark each tag as ✅ Safe, ⚠ Restricted, or ❌ Banned
javascript 
// ── INSTATOOL: Banned Hashtag Checker ──────────────────────
const HASHTAGS = [
  "photography", "fitness", "love", "instagood", "tbt"
  // ← add your hashtags here
];

(async () => {
  const sleep = ms => new Promise(r => setTimeout(r, ms));
  console.log(`%c🔍 Checking ${HASHTAGS.length} hashtags...`, "color:#00ff88");

  const results = { safe: [], restricted: [], banned: [] };

  for (const tag of HASHTAGS) {
    try {
      const res = await fetch(`https://www.instagram.com/api/v1/tags/${tag}/info/`).then(r => r.json());
      if (!res.id) {
        results.banned.push(tag);
        console.log(`%c❌ BANNED     #${tag}`, "color:#ff4466");
      } else if (res.is_restricted_use) {
        results.restricted.push(tag);
        console.log(`%c⚠  RESTRICTED #${tag}`, "color:#ffcc00");
      } else {
        results.safe.push(tag);
        console.log(`%c✅ SAFE       #${tag}  (${(res.media_count || 0).toLocaleString()} posts)`, "color:#00ff88");
      }
    } catch { results.banned.push(tag); }
    await sleep(800);
  }

  console.log(`%c
━━━━━━━━━━━━━━━━━━━━━━━━
SUMMARY
━━━━━━━━━━━━━━━━━━━━━━━━
✅ Safe:       ${results.safe.length}
⚠  Restricted: ${results.restricted.length}
❌ Banned:     ${results.banned.length}

Avoid these: ${[...results.banned, ...results.restricted].map(t => '#' + t).join(', ') || 'none'}
━━━━━━━━━━━━━━━━━━━━━━━━`, "color:#ffcc00;font-weight:bold");
})();

// RULES TO FOLLOW
01
Never run two action scripts at the same time. Open one tool, let it finish, then move to the next. Parallel actions multiply your rate and trigger Instagram's bot detection faster.
02
New accounts are high-risk. Accounts under 3 months old are flagged far more aggressively. If your account is new, stick to read-only tools only for the first month.
03
Wait between sessions. If you run the follow tool once, wait at least 1 hour before running it again — even if you didn't hit the limit. Instagram tracks patterns over time, not just single sessions.
04
If you get an action block, stop immediately. A temporary block (24–48h) is a warning. Keep pushing through it and it escalates to a permanent ban. Just wait it out.
05
Don't use this on your main account carelessly. These tools are powerful. Test with a secondary account first to understand the behavior before running on your primary.
06
You are fully responsible. IGNITX runs in your own browser using your own session. No data is sent anywhere. But Instagram can still detect patterns — use common sense.
