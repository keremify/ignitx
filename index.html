  const csrf   = document.cookie.match(/csrftoken=([^;]+)/)?.[1];
  if (!userId) { IX.st('❌ Not logged in','c-err'); return; }
  const H = {"x-ig-app-id":"936619743392459","x-csrftoken":csrf,"x-requested-with":"XMLHttpRequest","x-ig-www-claim":sessionStorage.getItem("www-claim-v2")||"0"};
  const PH = {...H,"content-type":"application/x-www-form-urlencoded"};
  const sleep = ms => new Promise(r => setTimeout(r, ms));

  const getList = async type => {
    let list=[], after="", more=true;
    while (more) {
      IX.st(`⏳ Loading ${type}... (${list.length} so far)`);
      const res = await fetch(`https://www.instagram.com/api/v1/friendships/${userId}/${type}/?count=200${after?"&max_id="+after:""}`,{headers:H}).then(r=>r.json());
      if (res.status==='login_required'){IX.st('❌ Session expired — refresh Instagram','c-err');return[];}
      list.push(...(res.users||[])); more=!!res.next_max_id; after=res.next_max_id||"";
      await sleep(600);
    }
    return list;
  };

  IX.pb(20); const following = await getList('following');
  IX.pb(60); const followers = await getList('followers');
  IX.pb(100);

  const fSet = new Set(followers.map(u=>u.pk));
  const nfb  = following.filter(u=>!fSet.has(u.pk));

  if (!nfb.length) {
    IX.st('✅ Everyone follows you back!');
    IX.body('<div class="ix-empty"><div style="font-size:28px">🎉</div><div>ALL CLEAR — NO NON-FOLLOWERS</div></div>');
    return;
  }

  IX.st(`Found <b>${nfb.length}</b> non-followers — ${following.length} following, ${followers.length} followers`);
  IX.body(`<div class="ix-sec">${nfb.length} ACCOUNTS NOT FOLLOWING YOU BACK</div>`);

  nfb.forEach(u => IX.add(`
    <div class="ix-urow" id="row${u.pk}">
      <input type="checkbox" class="ix-check ix-cb" data-pk="${u.pk}" checked>
      <div class="ix-av">${u.username[0]}</div>
      <div class="ix-uinfo">
        <div class="ix-uname">@${u.username}</div>
        <a class="ix-ulink" href="https://instagram.com/${u.username}" target="_blank">VISIT PROFILE ↗</a>
      </div>
      <button class="ix-abtn" id="ab${u.pk}" onclick="window._ixUnfOne('${u.pk}','${u.username}',this)">UNFOLLOW</button>
    </div>`));

  window._ixUnfOne = async (pk, username, btn) => {
    if (btn.classList.contains('done')) return;
    btn.textContent = '...'; btn.classList.add('done');
    await fetch(`https://www.instagram.com/api/v1/friendships/destroy/${pk}/`,{method:"POST",headers:PH});
    btn.textContent = '✓ DONE';
    const row = document.getElementById('row'+pk);
    if (row) { row.style.opacity='0.35'; row.querySelector('.ix-cb').checked=false; }
  };

  IX.foot(`<button class="ix-btn ix-br" onclick="window._ixUnfAll()">UNFOLLOW SELECTED</button>
           <button class="ix-btn ix-by" onclick="window._ixExport()">EXPORT CSV</button>`);

  window._ixUnfAll = async () => {
    const cbs = [...document.querySelectorAll('.ix-cb:checked')];
    if (!cbs.length) { IX.st('⚠ No accounts selected','c-warn'); return; }
    IX.st(`Unfollowing ${cbs.length} accounts...`,'c-warn');
    for (const cb of cbs) {
      const btn = document.getElementById('ab'+cb.dataset.pk);
      await window._ixUnfOne(cb.dataset.pk, '', btn);
      await sleep(3600);
    }
    IX.st('✅ Done unfollowing!');
  };

  window._ixExport = () => {
    const csv = 'username,profile_url\n' + nfb.map(u=>`@${u.username},https://instagram.com/${u.username}`).join('\n');
    const a = Object.assign(document.createElement('a'),{href:URL.createObjectURL(new Blob([csv],{type:'text/csv'})),download:'non_followers.csv'});
    document.body.appendChild(a); a.click(); a.remove();
    IX.st('✅ CSV downloaded!');
  };
})();
